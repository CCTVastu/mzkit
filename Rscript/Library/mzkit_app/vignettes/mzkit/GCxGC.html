let extract_2D_peaks as function(raw as  mzPack, 
mz as double = NAN, 
mzdiff as any = "ppm:30", 
[<Environment>]) -> <generic> D2Chromatogram {
    #
    # .NET API information
    #
    # module: mzkit.GCxGC
    # LibPath: D:/mzkit/Rscript/Library/mzkit_app/assembly/net6.0
    # library: mzkit.dll
    # package: "GCxGC"
    #
    return call R#.interop_[GCxGC::create2DPeaks](...);
}

Arguments

  • raw: -
  • mz: target mz value for extract XIC data. NA means extract
    TIC data by default.
  • mzdiff: the mz tolerance error for match the intensity data for
    extract XIC data if the mz is not
    NA value.

Details

this function will extract the TIC data by default.
